💡 EXAMPLE USAGE
══════════════════════════════════════════════════════════
import * as allure from 'allure-js-commons';

test.describe('My Tests', () => {
  test.beforeEach(async () => {
    await allure.epic('E-commerce');
    await allure.feature('Shopping Cart');
  });

  test('Add product', async ({ page }) => {
    await allure.story('Product Addition');
    await allure.severity('critical');
    await allure.description('Verify product can be added');
    
    // Test code with steps
    await test.step('Add product', async () => {
      // actions
    });
    
    // Screenshot attachment
    const screenshot = await page.screenshot({ fullPage: true });
    await allure.attachment('Screenshot', screenshot, 'image/png');
  });
});
